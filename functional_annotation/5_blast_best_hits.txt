#!/bin/bash

# FILTER BEST BLAST HITS
# This script processes BLAST output .tsv files, filters hits for target genes,
# selects the best hit per gene based on percent identity (>=90%) and bit score,
# and stores the results in a separate directory.

# Directories and files
INPUT_DIR="./blast_output"             # BLAST output .tsv files
OUTPUT_DIR="./blast_best_hits"         # Directory to store filtered hits
TARGET_GENES="./target_genes.txt"      # File containing target genes

# Create output directory if it doesn't exist
mkdir -p "$OUTPUT_DIR"

# Loop over each BLAST file
for f in "$INPUT_DIR"/*.tsv; do
    # Skip if no TSV files
    [ -e "$f" ] || continue

    # Extract base file name
    base_name=$(basename "$f" .tsv)
    output="$OUTPUT_DIR/${base_name}.tsv"

    echo "Processing $f -> $output"

    # AWK script to filter best hits
    awk -F'\t' -v OFS="\t" -v genes_file="$TARGET_GENES" '
    BEGIN {
        # Load target genes into an array
        while ((getline line < genes_file) > 0) target[line] = 1
    }
    {
        # Extract gene name from subject ID
        split($2, a, "_")
        gene = a[1]
        identity = $3
        bit = $12

        # Skip genes not in target list
        if (!(gene in target)) next

        # Filter best hit based on >=90% identity
        if(identity >= 90) {
            if(!(gene in best) || identity > best[gene] || (identity == best[gene] && bit > best_bit[gene])) {
                best[gene] = identity
                best_bit[gene] = bit
                line_best[gene] = $0
            }
        } else {
            # Store first hit if no >=90% hit exists
            if(!(gene in first_hit)) first_hit[gene] = $0
        }
    }
    END {
        # Print best hits first
        for(g in line_best) print line_best[g]
        # Print fallback hits for genes without >=90% hits
        for(g in first_hit) if(!(g in line_best)) print first_hit[g]
    }
    ' "$f" > "$output"
done

echo "All files processed. Filtered hits are in $OUTPUT_DIR"
